14. Как сравнить два объекта(поверхностная копия)

    1. Сравниваем кол-во ключей

    2. Идем по каждому ключу первого объекта
        Проверяем:
            1. есть ли такой ключ во втором объекте
            2. равны ли значения по ключу.
                Здесь проверяем через !Object.is()

    ////////////
    const keys1 = Object.keys(obj1)
    const keys2 = Object.keys(obj2)

    if(keys1.length !== keys2.length) {
        return false
    }

    for(const key of keys1) {
        if(!keys2.includes(key) || !Object.is(obj1[key], obj2[key])) {
            return false
        }
    }

    return true
