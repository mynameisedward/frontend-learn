------------- ------------- Методы массивов ------------- ------------- 

HOF(High order functions) - это функция которая принимает другую функцию в качестве аргумента
                                либо возвращает функцию

// .forEach()
    Применяет переданную функцию к каждому элементу массива
    arr.forEach(function(obj, index, arr) {
        // тело функции
    })
    Функция принимает 3 параметра:
        1. Каждый элемет массива
        2. Индекс
        3. Сам массив



//  .map()
    Принимает ту же самую функцию
    Применяет функцию к каждому элементу и возвращает новый массив с этими элементами

    const newArr = arr.map(elem => {
        // тело функции
    })


// .filter()
    Фильтрует массив по какому-либо условию
    Создает новый отфильтрованный массив
    Применяет ту же функцию

    const filterArr = arr.filter((elem) => {
        return elem > 2
    })
    


// .reduce()
    Получить одно значение на основе всех элементов массива
    Принимает 2 параметра:
        1. callback(total, elem)
        2. и начальное значение, присвоенное total

    const result = arr.reduce((total, elem) => {

    }, 0)



// .find()
    Находит элемент по условию
    Принимает callback
    
    const numb = arr.find((elem) => (elem / 2) == 3)



// .findIndex()
    Находит индекс по условию
    Принимает callback

    const idx = arr.find((elem) => elem === 17)
